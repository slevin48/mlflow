/**
 * Use ffmpeg scene change detection to find segments of activity.
 * Returns array of { start, end, type: 'active'|'idle' }
 */
function detectScenes(inputFile, duration) {
  // Use ffprobe to detect scene changes above threshold
  const sceneThreshold = 0.01;
  const out = run(
    `"${FFPROBE}" -v error -show_entries frame=pts_time -of csv=p=0 ` +
    `-select_streams v:0 -vf "select='gt(scene,${sceneThreshold})'" "${inputFile}"`
  );

  if (!out || out.trim() === '') {
    // No scene changes detected - treat whole video as idle with brief active bookends
    return [
      { start: 0, end: Math.min(2, duration), type: 'active' },
      { start: 2, end: Math.max(2, duration - 1), type: 'idle' },
      { start: Math.max(2, duration - 1), end: duration, type: 'active' },
    ];
  }

  const sceneTimes = out.trim().split('\n').map(t => parseFloat(t.trim())).filter(t => !isNaN(t));

  // Build segments: group scene changes within 2s as "active", gaps > 3s as "idle"
  const segments = [];
  let lastEnd = 0;
  const ACTIVE_WINDOW = 2.0;  // seconds around a scene change to keep at normal speed
  const IDLE_THRESHOLD = 3.0; // gap before we consider it idle

  for (const t of sceneTimes) {
    const activeStart = Math.max(lastEnd, t - ACTIVE_WINDOW / 2);
    const activeEnd = t + ACTIVE_WINDOW / 2;

    // If there's a gap before this active window, it's idle
    if (activeStart - lastEnd > IDLE_THRESHOLD) {
      segments.push({ start: lastEnd, end: activeStart, type: 'idle' });
    } else if (activeStart > lastEnd) {
      // Small gap - merge into active
      segments.push({ start: lastEnd, end: activeStart, type: 'active' });
    }

    segments.push({ start: activeStart, end: Math.min(activeEnd, duration), type: 'active' });
    lastEnd = Math.min(activeEnd, duration);
  }

  // Trailing idle
  if (lastEnd < duration - 0.5) {
    segments.push({ start: lastEnd, end: duration, type: 'idle' });
  }

  // Merge consecutive same-type segments
  const merged = [segments[0]];
  for (let i = 1; i < segments.length; i++) {
    const last = merged[merged.length - 1];
    if (segments[i].type === last.type) {
      last.end = segments[i].end;
    } else {
      merged.push(segments[i]);
    }
  }

  return merged;
}

/**
 * Process a single video: speed up idle parts, add text overlay
 */
function processVideo(entry, outputFile) {
  const inputFile = path.join(VID_DIR, entry.file);
  if (!fs.existsSync(inputFile)) {
    console.log(`  SKIP: ${entry.file} not found`);
    return false;
  }

  const duration = getDuration(inputFile);
  if (duration < 1) {
    console.log(`  SKIP: ${entry.file} too short (${duration}s)`);
    return false;
  }

  console.log(`  Detecting scenes in ${entry.file} (${duration.toFixed(1)}s)...`);
  const segments = detectScenes(inputFile, duration);

  const activeTotal = segments.filter(s => s.type === 'active').reduce((a, s) => a + (s.end - s.start), 0);
  const idleTotal = segments.filter(s => s.type === 'idle').reduce((a, s) => a + (s.end - s.start), 0);
  console.log(`  Active: ${activeTotal.toFixed(1)}s, Idle: ${idleTotal.toFixed(1)}s (${segments.length} segments)`);

  // Build ffmpeg filter: speed up idle segments 4x, keep active at 1x
  // We use trim + setpts to handle segments, then concat
  const persona = PERSONAS[entry.persona];
  const flowLabel = `Flow ${entry.flow}: ${entry.label}`;
  const nameLabel = `${persona.fullName} — ${persona.role}`;

  // For simplicity with variable speed: create a select filter expression
  // that applies different PTS multipliers based on timestamp ranges
  //
  // Approach: use segment-based trim, apply setpts, then concat
  const filterParts = [];
  const streamLabels = [];

  segments.forEach((seg, i) => {
    const speed = seg.type === 'idle' ? 4.0 : 1.0;
    const ptsExpr = `PTS-STARTPTS/${speed}`;  // N/speed for audio, 1/speed for video PTS
    const vLabel = `[v${i}]`;
    const aLabel = `[a${i}]`;

    filterParts.push(
      `[0:v]trim=start=${seg.start.toFixed(3)}:end=${seg.end.toFixed(3)},setpts=(PTS-STARTPTS)/${speed}${vLabel}`
    );
    streamLabels.push(vLabel);
  });

  // Concat all segments
  const concatInput = streamLabels.join('');
  filterParts.push(
    `${concatInput}concat=n=${streamLabels.length}:v=1:a=0[trimmed]`
  );

  // Add text overlay on concatenated result
  const escapedName = nameLabel.replace(/'/g, "'\\''");
  const escapedFlow = flowLabel.replace(/'/g, "'\\''");
  filterParts.push(
    `[trimmed]drawtext=text='${escapedName}':fontsize=20:fontcolor=white:` +
    `borderw=2:bordercolor=black:x=20:y=20,` +
    `drawtext=text='${escapedFlow}':fontsize=16:fontcolor=white:` +
    `borderw=2:bordercolor=black:x=20:y=48[out]`
  );

  const filterComplex = filterParts.join(';\n');

  // Write filter to temp file (complex filters are too long for command line)
  const filterFile = path.join(OUT_DIR, `filter_${entry.persona}_${entry.flow}.txt`);
  fs.writeFileSync(filterFile, filterComplex);

  const cmd = `"${FFMPEG}" -y -i "${inputFile}" -filter_complex_script "${filterFile}" ` +
    `-map "[out]" -c:v libx264 -preset fast -crf 23 -pix_fmt yuv420p "${outputFile}"`;

  const result = run(cmd, { timeout: 120000 });

  // Clean up filter file
  try { fs.unlinkSync(filterFile); } catch {}

  if (result !== null || fs.existsSync(outputFile)) {
    const outDur = getDuration(outputFile);
    console.log(`  Output: ${path.basename(outputFile)} (${outDur.toFixed(1)}s, was ${duration.toFixed(1)}s)`);
    return true;
  }
  return false;
}

/**
 * Concatenate multiple clips into one per-persona highlight reel
 */
function concatenatePersona(personaName, clipFiles, outputFile) {
  if (clipFiles.length === 0) return false;
  if (clipFiles.length === 1) {
    fs.copyFileSync(clipFiles[0], outputFile);
    console.log(`  Single clip, copied to ${path.basename(outputFile)}`);
    return true;
  }

  // Write concat list
  const listFile = path.join(OUT_DIR, `concat_${personaName}.txt`);
  const listContent = clipFiles.map(f => `file '${f.replace(/\\/g, '/')}'`).join('\n');
  fs.writeFileSync(listFile, listContent);

  const cmd = `"${FFMPEG}" -y -f concat -safe 0 -i "${listFile}" ` +
    `-c:v libx264 -preset fast -crf 23 -pix_fmt yuv420p "${outputFile}"`;

  run(cmd, { timeout: 120000 });

  try { fs.unlinkSync(listFile); } catch {}

  if (fs.existsSync(outputFile)) {
    const dur = getDuration(outputFile);
    console.log(`  Final reel: ${path.basename(outputFile)} (${dur.toFixed(1)}s)`);
    return true;
  }
  return false;
}
